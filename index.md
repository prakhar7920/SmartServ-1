    <script>
      const api_url = "https://s3.amazonaws.com/open-to-cors/assignment.json";
      async function getData() {
        const response = await fetch(api_url);
        const data = await response.json();
        const products = data.products;

        const array = [];
        const obj = JSON.parse(JSON.stringify(products));

        for (var i in obj) {
          array.push(obj[i]);
        }

        // sort on the basis of popularity
        const sorter = (a, b) => b.popularity - a.popularity;
        array.sort(sorter);

        document.write("<br>");
        for (var j = 0; j < 1000; j++) {
          document.write(
            "Category :- " +
              array[j].subcategory.toUpperCase().bold() +
              " ------ " +
              " Model :- " +
              array[j].title.bold() +
              " ------ " +
              " Price :- " +
              array[j].price.bold() +
              "/-" +
              " ------ " +
              " Popularity :- " +
              array[j].popularity.bold() +
              "</br>"
          );
          document.write("<br>");
        }

        console.log(array);
      }
      getData();
    </script>
